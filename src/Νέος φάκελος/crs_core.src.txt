isError = function(value)
  return value isa string and value != ""
end function

isApiSuccess = function(value)
  if value == null then return 0
  if isError(value) then return 0
  if value == 0 then return 0
  return 1
end function

normalizeInput = function(value, defaultValue = "")
  if value == null then return defaultValue
  return str(value)
end function

norm = function(value, defaultValue = "")
  return normalizeInput(value, defaultValue).trim
end function

normLower = function(value, defaultValue = "")
  return norm(value, defaultValue).lower
end function

normalizeText = function(value, defaultValue = "")
  clean = ""

  for ch in normalizeInput(value, defaultValue)
    if ch != CR then clean = clean + ch
  end for

  return clean
end function

toYesNoFlag = function(inputValue, defaultValue = 1)
  lowered = normLower(inputValue)
  if lowered == "" then return defaultValue
  if lowered == "y" or lowered == "yes" or lowered == "1" or lowered == "true" then return 1
  if lowered == "n" or lowered == "no" or lowered == "0" or lowered == "false" then return 0
  return defaultValue
end function

getParamValue = function(flagName, defaultValue = "")
  if params == null then return defaultValue

  flagIndex = params.indexOf(flagName)
  if flagIndex == null then return defaultValue

  nextIndex = flagIndex + 1
  if not params.hasIndex(nextIndex) then return defaultValue
  nextValue = norm(params[nextIndex])
  if nextValue.indexOf("--") == 0 then return defaultValue
  return params[nextIndex]
end function

hasParamFlag = function(flagName)
  if params == null then return 0
  return params.indexOf(flagName) != null
end function

getParamFlag = function(flagName, defaultValue = 0)
  if not hasParamFlag(flagName) then return defaultValue

  rawValue = getParamValue(flagName, "")
  if norm(rawValue) == "" then return 1
  return toYesNoFlag(rawValue, 1)
end function

normalizeInteger = function(inputValue, defaultValue = 0, minValue = null, maxValue = null)
  text = norm(inputValue)
  if not isNumericText(text) then return defaultValue

  parsed = text.val
  if minValue != null and parsed < minValue then parsed = minValue
  if maxValue != null and parsed > maxValue then parsed = maxValue
  return parsed
end function

getParamInt = function(flagName, defaultValue = 0, minValue = null, maxValue = null)
  return normalizeInteger(getParamValue(flagName, "" + defaultValue), defaultValue, minValue, maxValue)
end function

isNumericText = function(textValue)
  text = norm(textValue)
  if text == "" then return 0

  for ch in text
    if ch < "0" or ch > "9" then return 0
  end for

  return 1
end function

listHasValue = function(items, wanted)
  if items == null then return 0

  for item in items
    if item == wanted then return 1
  end for

  return 0
end function

pushUniqueString = function(values, seen, rawValue)
  valueText = norm(rawValue)
  if valueText == "" then return 0
  if listHasValue(seen, valueText) then return 0

  seen.push(valueText)
  values.push(valueText)
  return 1
end function

ensureFolder = function(computer, parentPath, folderName)
  folderPath = parentPath + "/" + folderName
  folder = computer.File(folderPath)
  if folder != null then
    if not folder.is_folder then return null
    if not folder.has_permission("w") then return null
    return folderPath
  end if

  created = computer.create_folder(parentPath, folderName)
  if isError(created) then return null

  folder = computer.File(folderPath)
  if folder == null then return null
  if not folder.is_folder then return null
  if not folder.has_permission("w") then return null
  return folderPath
end function

end function

extractUnsafeValues = function(scanInfo)
  values = []
  seen = []
  remaining = normalizeInput(scanInfo)

  while 1
    tagStart = remaining.indexOf("<b>")
    if tagStart == null then break

    remaining = remaining[tagStart + 3:]
    tagEnd = remaining.indexOf("</b>")
    if tagEnd == null then break

    candidate = remaining[0:tagEnd]
    if candidate != "" and not listHasValue(seen, candidate) then
      seen.push(candidate)
      values.push(candidate)
    end if

    if tagEnd + 4 >= remaining.len then break
    remaining = remaining[tagEnd + 4:]
  end while

  if values.len == 0 then
    for rawLine in normalizeText(scanInfo).split(NL)
      cleanLine = norm(rawLine)
      if cleanLine == "" then continue
      if cleanLine.lower.indexOf("unchecked") == null then continue

      candidate = cleanLine.replace("<b>", "").replace("</b>", "").replace(CR, "")
      candidate = norm(candidate)
      if candidate == "" then continue
      if listHasValue(seen, candidate) then continue

      seen.push(candidate)
      values.push(candidate)
    end for
  end if

  return values
end function

loadSignatureCache = function(cacheFile)
  cacheData = {
    "records": [],
    "lines": [],
    "keys": [],
  }

  if cacheFile == null then return cacheData
  if not cacheFile.has_permission("r") then return cacheData

  content = normalizeInput(cacheFile.get_content)
  if content == "" then return cacheData

  for line in content.split(NL)
    cleanLine = normalizeInput(line).trim
    if cleanLine == "" then continue

    parts = cleanLine.split(TAB)
    if parts == null or parts.len < 6 then continue
    if parts[0] == "lib" and parts[1] == "version" then continue

    signatureKey = buildCacheSignatureKey(parts[0], parts[1], parts[2], parts[3], parts[4])
    if listHasValue(cacheData.keys, signatureKey) then continue

    cacheData.keys.push(signatureKey)
    cacheData.lines.push(cleanLine)
    cacheData.records.push({
      "lib": parts[0],
      "version": parts[1],
      "port": parts[2],
      "memory": parts[3],
      "unsafe": parts[4],
      "type": parts[5],
    })
  end for

  return cacheData
end function

saveSignatureCache = function(cacheFile, cacheData)
  if cacheFile == null then return "Missing signature cache file"

  lines = ["lib" + TAB + "version" + TAB + "port" + TAB + "memory" + TAB + "unsafe" + TAB + "type"]
  if cacheData != null and cacheData.lines != null then
    for cacheLine in cacheData.lines
      appendLine(lines, cacheLine)
    end for
  end if

  return cacheFile.set_content(lines.join(NL))
end function

end function

findCachedSignatures = function(cacheData, libName, libVersion, portNumber)
  matches = []
  if cacheData == null or cacheData.records == null then return matches

  wantedLib = normalizeInput(libName).trim
  wantedVersion = normalizeInput(libVersion).trim
  wantedPort = normalizeInput(portNumber).trim

  for record in cacheData.records
    if record == null then continue
    if record.lib == wantedLib and record.version == wantedVersion and record.port == wantedPort then
      matches.push(record)
    end if
  end for

  if matches.len > 0 then return matches

  for record in cacheData.records
    if record == null then continue
    if record.lib == wantedLib and record.version == wantedVersion then
      matches.push(record)
    end if
  end for

  return matches
end function

normalizeCacheField = function(value)
  clean = ""

  for ch in normalizeText(value).trim
    if ch == TAB or ch == NL then
      clean = clean + " "
    else
      clean = clean + ch
    end if
  end for

  return clean
end function

loadWorkerManifest = function(manifestFile)
  manifest = {
    "mode": "",
    "status": "",
    "accepted_host": "",
    "accepted_access_level": "",
    "selected_host": "",
    "target_ip": "",
    "target_lan_ip": "",
    "target_user": "",
    "signature_records": 0,
    "credential_records": 0,
    "surface_records": 0,
    "loot_records": 0,
    "visited_hosts": [],
    "visited_host_keys": {},
  }

  if manifestFile == null then return manifest
  if not manifestFile.has_permission("r") then return manifest

  content = normalizeInput(manifestFile.get_content)
  if content == "" then return manifest

  for line in content.split(NL)
    cleanLine = norm(line)
    if cleanLine == "" then continue

    parts = cleanLine.split(TAB)
    if parts == null or parts.len < 2 then continue

    fieldName = normLower(parts[0])
    fieldValue = parts[1]
    if fieldName == "field" then continue

    if fieldName == "visited_host" then
      if manifest.visited_host_keys.hasIndex(fieldValue) then continue
      manifest.visited_host_keys[fieldValue] = 1
      manifest.visited_hosts.push(fieldValue)
      continue
    end if

    if fieldName == "mode" then manifest.mode = fieldValue
    if fieldName == "status" then manifest.status = fieldValue
    if fieldName == "accepted_host" then manifest.accepted_host = fieldValue
    if fieldName == "accepted_access_level" then manifest.accepted_access_level = fieldValue
    if fieldName == "selected_host" then manifest.selected_host = fieldValue

    if fieldName == "surface_records" then manifest.surface_records = normalizeInteger(fieldValue, 0, 0)

writeWorkerManifest = function(outputFiles, runtimeContext, runState, finalResult = null)
  if outputFiles == null then return "Missing output files"
  if outputFiles.worker_manifest_file == null then return "Missing worker manifest file"

  signatureCount = 0
  credentialCount = 0
  surfaceCount = 0
  lootCount = 0
  selectedHost = ""
  statusText = ""
  acceptedHost = ""
  acceptedAccessLevel = ""

  if runState != null then
    if runState.signature_cache != null and runState.signature_cache.records != null then signatureCount = runState.signature_cache.records.len
    if runState.credential_cache != null and runState.credential_cache.records != null then credentialCount = runState.credential_cache.records.len
    if runState.surface_records != null then surfaceCount = runState.surface_records.len
    if runState.loot_records != null then lootCount = runState.loot_records.len
    selectedHost = norm(runState.selected_host_ip)
  end if

  if finalResult != null then
    statusText = norm(finalResult.status)
    acceptedHost = norm(finalResult.accepted_host)
    acceptedAccessLevel = norm(finalResult.accepted_access_level)
  end if

  lines = ["field" + TAB + "value"]
  appendLine(lines, "build" + TAB + SCRIPT_BUILD)
  if runtimeContext != null then appendLine(lines, "date" + TAB + formatValueDisplay(runtimeContext.current_date))
  if runState != null and runState.config != null then appendLine(lines, "mode" + TAB + formatValueDisplay(runState.config.mode))
  appendLine(lines, "status" + TAB + formatValueDisplay(statusText))
  appendLine(lines, "accepted_host" + TAB + formatValueDisplay(acceptedHost))
  appendLine(lines, "accepted_access_level" + TAB + formatValueDisplay(acceptedAccessLevel))
  appendLine(lines, "selected_host" + TAB + formatValueDisplay(selectedHost))
  if runState != null and runState.target != null then
    appendLine(lines, "target_ip" + TAB + formatValueDisplay(runState.target.target_ip))
    appendLine(lines, "target_lan_ip" + TAB + formatValueDisplay(runState.target.target_lan_ip))
    appendLine(lines, "target_user" + TAB + formatValueDisplay(runState.target.target_user))
  end if
  appendLine(lines, "signature_records" + TAB + signatureCount)
  appendLine(lines, "credential_records" + TAB + credentialCount)
  appendLine(lines, "surface_records" + TAB + surfaceCount)
  appendLine(lines, "loot_records" + TAB + lootCount)

  if runState != null and runState.visited_hosts != null then
    for visitedHost in runState.visited_hosts
      if visitedHost == null then continue
      appendLine(lines, "visited_host" + TAB + norm(visitedHost))
    end for
  end if

  return outputFiles.worker_manifest_file.set_content(lines.join(NL))
end function

makePortInfo = function(portNumber)
  return {
    "port": portNumber,
    "service": serviceName(portNumber),
  }

pushOpenPort = function(openPorts, seen, portNumber)
  portKey = "" + portNumber
  if listHasValue(seen, portKey) then return 0
  seen.push(portKey)
  openPorts.push(makePortInfo(portNumber))
  return 1
end function

collectRouterEntryPorts = function(router)
  openPorts = []
  seen = []
  if router == null then return openPorts

  pushOpenPort(openPorts, seen, 0)

  usedPorts = router.used_ports
  if usedPorts == null then return openPorts

  for portObj in usedPorts
    if portObj == null then continue
    if portObj.is_closed then continue
    pushOpenPort(openPorts, seen, portObj.port_number)
  end for

  return openPorts
end function

collectOpenPorts = function(router, ipAddress)
  openPorts = []
  seen = []
  if router == null then return openPorts

  ports = router.device_ports(ipAddress)
  if ports == null then return openPorts

  for portObj in ports
    if portObj == null then continue
    if portObj.is_closed then continue
    pushOpenPort(openPorts, seen, portObj.port_number)
  end for

  return openPorts
end function

scoreHost = function(ipAddress, openPorts)
  score = 0
  if isLikelyRouterLanIp(ipAddress) then score = score + 1
  if openPorts == null then return score

  for portInfo in openPorts
    if portInfo == null then continue
    score = score + serviceWeight(portInfo.port)
  end for

  return score
end function

hostRoleGuess = function(ipAddress, openPorts)
  if isLikelyRouterLanIp(ipAddress) then
    if openPorts == null or openPorts.len == 0 then return "adjacent-gateway"
    return "gateway"
  end if

  if openPorts == null or openPorts.len == 0 then return "host"

  for portInfo in openPorts
    if portInfo.port == 5555 then return "smartphone"
    if portInfo.port == 3306 or portInfo.port == 5432 or portInfo.port == 141 then return "data-server"
    if portInfo.port == 80 or portInfo.port == 443 or portInfo.port == 8080 then return "web-server"
    if portInfo.port == 25 then return "mail-server"
    if portInfo.port == 1883 then return "iot-device"
    if portInfo.port == 1542 then return "repository-server"
  end for

  return "host"
end function

makeHostRecord = function(ipAddress, roleText, priorityText, scoreValue, openPorts, sourceText, publicIp = "", lanIp = "", gatewayIp = "")
  hostIp = normalizeInput(ipAddress).trim
  normalizedPublicIp = normalizeInput(publicIp).trim
  normalizedLanIp = normalizeInput(lanIp).trim
  normalizedGatewayIp = normalizeInput(gatewayIp).trim
  normalizedRole = normalizeInput(roleText).trim
  normalizedSource = normalizeInput(sourceText).trim

  if normalizedRole == "" then normalizedRole = "host"
  if normalizedSource == "" then normalizedSource = "discovery"
  if normalizedPublicIp == "" and not is_lan_ip(hostIp) then normalizedPublicIp = hostIp
  if normalizedLanIp == "" and is_lan_ip(hostIp) then normalizedLanIp = hostIp
  if priorityText == "" then priorityText = priorityLabel(scoreValue)

  return {
    "ip": hostIp,
    "public_ip": normalizedPublicIp,
    "lan_ip": normalizedLanIp,
    "gateway_ip": normalizedGatewayIp,
    "role": normalizedRole,
    "priority": normalizeInput(priorityText).trim.lower,
    "score": scoreValue,
    "open_ports": openPorts,
    "source": normalizedSource,
  }
end function

addUniqueHostRecord = function(hostRecords, seen, hostRecord)
  if hostRecord == null then return 0

  hostKey = normalizeInput(hostRecord.ip).trim
  if hostKey == "" then return 0
  if listHasValue(seen, hostKey) then return 0

  seen.push(hostKey)
  hostRecords.push(hostRecord)
  return 1
end function

formatOpenPorts = function(openPorts)
  if openPorts == null or openPorts.len == 0 then return "none"

  parts = []
  for portInfo in openPorts
    if portInfo == null then continue
    parts.push(portInfo.port + " (" + serviceName(portInfo.port) + ")")
  end for

  if parts.len == 0 then return "none"
  return parts.join(", ")
end function

formatPortNumbers = function(openPorts)
  if openPorts == null or openPorts.len == 0 then return "none"

  parts = []
  for portInfo in openPorts
    if portInfo == null then continue
    parts.push("" + portInfo.port)
  end for

  if parts.len == 0 then return "none"
  return parts.join(",")
end function

routerHasLanIp = function(router, ipAddress)
  if router == null then return 0
  wantedIp = normalizeInput(ipAddress).trim
  if wantedIp == "" then return 0
  if normalizeInput(router.local_ip).trim == wantedIp then return 1

  lanIps = router.devices_lan_ip
  if lanIps == null then return 0

  for lanIp in lanIps
    if normalizeInput(lanIp).trim == wantedIp then return 1
  end for

  return 0
end function

buildRouterSelfRecord = function(router, usePublicIp = 1, sourceText = "router-discovery")
  if router == null then return null

  primaryIp = normalizeInput(router.local_ip).trim
  if usePublicIp and normalizeInput(router.public_ip).trim != "" then primaryIp = normalizeInput(router.public_ip).trim

  openPorts = collectRouterEntryPorts(router)
  scoreValue = scoreHost(primaryIp, openPorts) + 3
  return makeHostRecord(primaryIp, "gateway", priorityLabel(scoreValue), scoreValue, openPorts, sourceText, normalizeInput(router.public_ip).trim, normalizeInput(router.local_ip).trim, normalizeInput(router.local_ip).trim)
end function

buildRouterHostRecords = function(router, preferredIp = "", usePublicEntry = 1, sourceText = "router-discovery")
  hostRecords = []
  seen = []
  if router == null then return hostRecords

  addUniqueHostRecord(hostRecords, seen, buildRouterSelfRecord(router, usePublicEntry, sourceText))

  lanIps = router.devices_lan_ip
  if lanIps != null then
    for lanIp in lanIps
      openPorts = collectOpenPorts(router, lanIp)
      scoreValue = scoreHost(lanIp, openPorts)
      roleText = hostRoleGuess(lanIp, openPorts)
      addUniqueHostRecord(hostRecords, seen, makeHostRecord(lanIp, roleText, priorityLabel(scoreValue), scoreValue, openPorts, sourceText, normalizeInput(router.public_ip).trim, normalizeInput(lanIp).trim, normalizeInput(router.local_ip).trim))
    end for
  end if

  normalizedPreferredIp = normalizeInput(preferredIp).trim
  if normalizedPreferredIp != "" and not listHasValue(seen, normalizedPreferredIp) then
    openPorts = collectOpenPorts(router, normalizedPreferredIp)
    scoreValue = scoreHost(normalizedPreferredIp, openPorts)
    roleText = hostRoleGuess(normalizedPreferredIp, openPorts)
    addUniqueHostRecord(hostRecords, seen, makeHostRecord(normalizedPreferredIp, roleText, priorityLabel(scoreValue), scoreValue, openPorts, sourceText + "-preferred", normalizeInput(router.public_ip).trim, normalizedPreferredIp, normalizeInput(router.local_ip).trim))
  end if

  return hostRecords
end function

parseHostPortsText = function(textValue)
  openPorts = []
  seen = []
  cleanText = normalizeInput(textValue).trim
  if cleanText == "" or cleanText.lower == "none" then return openPorts

  for rawPart in cleanText.split(",")
    part = normalizeInput(rawPart).trim
    if part == "" then continue

    portText = part
    if part.indexOf(" ") != null then
      splitParts = part.split(" ")
      if splitParts != null and splitParts.len > 0 then portText = splitParts[0]
    end if
    if not isNumericText(portText) then continue

    pushOpenPort(openPorts, seen, portText.val)
  end for

  return openPorts
end function

parseMapperHostLine = function(lineText)
  cleanLine = normalizeInput(lineText).trim
  if cleanLine == "" then return null

  if cleanLine.indexOf("|") == null then return null
  parts = cleanLine.split("|")
  if parts == null or parts.len < 2 then return null

  ipAddress = normalizeInput(parts[0]).trim
  if not isValidIpAddress(ipAddress) then return null

  roleText = normalizeInput(parts[1]).trim
  priorityText = ""
  scoreValue = 0
  portsText = "none"

  for part in parts
    cleanedPart = normalizeInput(part).trim
    lowered = cleanedPart.lower
    if cleanedPart == "" then continue
    if cleanedPart == ipAddress then continue
    if cleanedPart == roleText then continue
    if lowered.indexOf("priority=") == 0 then
      splitParts = cleanedPart.split("=")
      if splitParts != null and splitParts.len > 1 then priorityText = splitParts[1]
      continue
    end if
    if lowered.indexOf("score=") == 0 then
      splitParts = cleanedPart.split("=")
      if splitParts != null and splitParts.len > 1 then
        scoreText = splitParts[1]
        if isNumericText(scoreText) then scoreValue = scoreText.val
      end if
      continue
    end if
    portsText = cleanedPart
  end for

  openPorts = parseHostPortsText(portsText)
  if scoreValue == 0 then scoreValue = scoreHost(ipAddress, openPorts)
  if priorityText == "" then priorityText = priorityLabel(scoreValue)
  return makeHostRecord(ipAddress, roleText, priorityText, scoreValue, openPorts, "mapper-file")
end function

loadMapperHostRecords = function(computer, mapperFilePath)
  hostRecords = []
  seen = []
  normalizedPath = normalizeInput(mapperFilePath).trim
  if normalizedPath == "" then return hostRecords

  mapperFile = computer.File(normalizedPath)
  mapperContent = readTextFile(mapperFile)
  if mapperContent == null or mapperContent == "" then return hostRecords

  for line in mapperContent.split(NL)
    hostRecord = parseMapperHostLine(line)
    addUniqueHostRecord(hostRecords, seen, hostRecord)
  end for

  return hostRecords
end function

discoverTargets = function(targetContext, runtimeContext, runState)
  discoveredHosts = []
  seen = []

  mappedHosts = loadMapperHostRecords(runtimeContext.computer, runState.config.mapper_file)
  for mappedHost in mappedHosts
    addUniqueHostRecord(discoveredHosts, seen, mappedHost)
  end for

  if targetContext.entry_type == "public_ip" then
    targetRouter = get_router(targetContext.entry_ip)
    if targetRouter != null then
      for hostRecord in buildRouterHostRecords(targetRouter, targetContext.target_lan_ip, 1, "public-router")
        addUniqueHostRecord(discoveredHosts, seen, hostRecord)
      end for
    else
      addUniqueHostRecord(discoveredHosts, seen, makeHostRecord(targetContext.entry_ip, "public-host", "medium", 4, [], "public-fallback"))
    end if
  else
    if targetContext.entry_type == "router_ip" then
      targetRouter = null
      if runtimeContext.current_router != null and routerHasLanIp(runtimeContext.current_router, targetContext.entry_ip) then
        targetRouter = runtimeContext.current_router
      end if

      if targetRouter == null and not is_lan_ip(targetContext.entry_ip) then
        targetRouter = get_router(targetContext.entry_ip)
      end if

      if targetRouter != null then
        for hostRecord in buildRouterHostRecords(targetRouter, targetContext.target_lan_ip, 0, "router-target")
          addUniqueHostRecord(discoveredHosts, seen, hostRecord)
        end for
      else
        fallbackScore = scoreHost(targetContext.entry_ip, []) + 3
        addUniqueHostRecord(discoveredHosts, seen, makeHostRecord(targetContext.entry_ip, "gateway", priorityLabel(fallbackScore), fallbackScore, [], "router-fallback", "", targetContext.entry_ip, targetContext.entry_ip))
      end if
    else
      if targetContext.entry_type == "lan_ip" then
        if runtimeContext.current_router != null and routerHasLanIp(runtimeContext.current_router, targetContext.entry_ip) then
          for hostRecord in buildRouterHostRecords(runtimeContext.current_router, targetContext.entry_ip, 0, "current-router")
            addUniqueHostRecord(discoveredHosts, seen, hostRecord)
          end for
        else
          addUniqueHostRecord(discoveredHosts, seen, makeHostRecord(targetContext.entry_ip, "host", "medium", scoreHost(targetContext.entry_ip, []), [], "lan-direct"))
        end if
      else
        if runtimeContext.current_router != null then
          for hostRecord in buildRouterHostRecords(runtimeContext.current_router, runtimeContext.computer.local_ip, 0, "current-router")
            addUniqueHostRecord(discoveredHosts, seen, hostRecord)
          end for
        else
          addUniqueHostRecord(discoveredHosts, seen, makeHostRecord(runtimeContext.computer.local_ip, "host", "medium", 1, [], "current-host", normalizeInput(runtimeContext.computer.public_ip).trim, normalizeInput(runtimeContext.computer.local_ip).trim))
        end if
      end if
    end if
  end if

  runState.discovered_hosts = discoveredHosts
  return discoveredHosts
end function

hostPriorityScore = function(hostRecord, targetContext, runState = null)
  if hostRecord == null then return -999999

  scoreValue = hostRecord.score * 10
  roleText = normalizeInput(hostRecord.role).trim.lower
  workerMode = 0
  currentHostPublicIp = ""
  currentHostLocalIp = ""
  currentRouterPublicIp = ""
  currentRouterLocalIp = ""

  if runState != null and runState.config != null then
    if normalizeInput(runState.config.mode).trim == "worker" then workerMode = 1
  end if

  if runState != null and runState.runtime != null and runState.runtime.computer != null then
    currentHostPublicIp = normalizeInput(runState.runtime.computer.public_ip).trim
    currentHostLocalIp = normalizeInput(runState.runtime.computer.local_ip).trim
  end if

  if runState != null and runState.runtime != null and runState.runtime.current_router != null then
    currentRouterPublicIp = normalizeInput(runState.runtime.current_router.public_ip).trim
    currentRouterLocalIp = normalizeInput(runState.runtime.current_router.local_ip).trim
  end if

  if normalizeInput(hostRecord.priority) == "high" then scoreValue = scoreValue + 20
  if roleText.indexOf("gateway") != null then scoreValue = scoreValue + 15
  if roleText == "smartphone" then scoreValue = scoreValue - 10

  if not workerMode and targetContext.entry_type == "public_ip" and targetContext.target_lan_ip == "" then
    if hostRecord.ip == targetContext.entry_ip then scoreValue = scoreValue + 120
    if normalizeInput(hostRecord.public_ip).trim == targetContext.entry_ip then scoreValue = scoreValue + 80
  end if

  if targetContext.entry_type == "router_ip" and hostRecord.ip == targetContext.entry_ip then scoreValue = scoreValue + 140
  if not workerMode and targetContext.target_ip != "" and (hostRecord.ip == targetContext.target_ip or normalizeInput(hostRecord.public_ip).trim == targetContext.target_ip) then scoreValue = scoreValue + 120
  if targetContext.target_lan_ip != "" and (hostRecord.ip == targetContext.target_lan_ip or normalizeInput(hostRecord.lan_ip).trim == targetContext.target_lan_ip) then scoreValue = scoreValue + 200

  if workerMode then
    if currentHostLocalIp != "" and (hostRecord.ip == currentHostLocalIp or normalizeInput(hostRecord.lan_ip).trim == currentHostLocalIp) then scoreValue = scoreValue - 300
    if currentHostLocalIp == "" and currentHostPublicIp != "" and normalizeInput(hostRecord.public_ip).trim == currentHostPublicIp then scoreValue = scoreValue - 300
    if currentRouterPublicIp != "" and normalizeInput(hostRecord.ip).trim == currentRouterPublicIp then scoreValue = scoreValue - 260
    if currentRouterPublicIp != "" and currentRouterLocalIp != "" and normalizeInput(hostRecord.public_ip).trim == currentRouterPublicIp and normalizeInput(hostRecord.lan_ip).trim == currentRouterLocalIp then scoreValue = scoreValue - 260
  end if

  return scoreValue
end function

rankTargets = function(hostRecords, targetContext, runState)
  rankedHosts = []
  used = []
  if hostRecords == null then return rankedHosts

  while used.len < hostRecords.len
    yield
    bestHost = null
    bestKey = ""
    bestScore = -999999

    for hostRecord in hostRecords
      if hostRecord == null then continue
      hostKey = normalizeInput(hostRecord.ip).trim
      if hostKey == "" then continue
      if listHasValue(used, hostKey) then continue

      hostScore = hostPriorityScore(hostRecord, targetContext, runState)
      if bestHost == null or hostScore > bestScore then
        bestHost = hostRecord
        bestKey = hostKey
        bestScore = hostScore
      else
        if hostScore == bestScore and hostRecord.ip < bestHost.ip then
          bestHost = hostRecord
          bestKey = hostKey
          bestScore = hostScore
        end if
      end if
    end for

    if bestHost == null then break
    used.push(bestKey)
    rankedHosts.push(bestHost)
  end while

  runState.ranked_hosts = rankedHosts
  return rankedHosts
end function

  if hostRecords == null then return null

  for hostRecord in hostRecords
    if hostRecord == null then continue

  end for

findExactTargetHostRecord = function(hostRecords, targetContext)
  if hostRecords == null then return null
  if targetContext == null then return null

  targetLanIp = normalizeInput(targetContext.target_lan_ip).trim
  if targetLanIp != "" then
    targetRecord = findHostRecordByIp(hostRecords, targetLanIp)
    if targetRecord != null then return targetRecord

    for hostRecord in hostRecords
      if hostRecord == null then continue
      if normalizeInput(hostRecord.lan_ip).trim == targetLanIp then return hostRecord
    end for
  end if

  targetIp = normalizeInput(targetContext.target_ip).trim
  if targetIp != "" then
    targetRecord = findHostRecordByIp(hostRecords, targetIp)
    if targetRecord != null then return targetRecord

    for hostRecord in hostRecords
      if hostRecord == null then continue
      if normalizeInput(hostRecord.public_ip).trim == targetIp then return hostRecord
    end for
  end if

  return null
end function

findPreferredEntryHostRecord = function(hostRecords, targetContext)
  if hostRecords == null then return null
  if targetContext == null then return null
  if not targetContext.exact_internal_target then return null
  if normalizeInput(targetContext.entry_type).trim.lower != "public_ip" then return null

  targetPublicIp = normalizeInput(targetContext.target_ip).trim
  gatewayFallback = null

  for hostRecord in hostRecords
    if hostRecord == null then continue

    roleText = normalizeInput(hostRecord.role).trim.lower
    if roleText.indexOf("gateway") == null then continue
    if gatewayFallback == null then gatewayFallback = hostRecord

    hostIp = normalizeInput(hostRecord.ip).trim
    hostPublicIp = normalizeInput(hostRecord.public_ip).trim
    if targetPublicIp != "" and (hostIp == targetPublicIp or hostPublicIp == targetPublicIp) then return hostRecord
  end for

  return gatewayFallback
end function

isExactTargetHost = function(runState, targetRecord)
  if runState == null then return 0
  if targetRecord == null then return 0

  exactTargetRecord = findExactTargetHostRecord(runState.ranked_hosts, runState.target)
  if exactTargetRecord == null then return 0

  return normalizeInput(exactTargetRecord.ip).trim == normalizeInput(targetRecord.ip).trim
end function
